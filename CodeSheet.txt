git config --global user.name "Tu nombre" --configuración inicial
git config --global user.email "Tu correo" --configuración inicial
git init --iniciar repositorio
git status --ver el estado del repositorio
git add Archivo --agregar archivo
git add . --agregar todos los archivos
git commit -m "Mensaje" --hacer commmit a los archivos agregados; es importante ser descriptivo con el Mensaje
git reset archivo-- quitar archivo del state
git reset .-- quitar archivo del state
git restore archivo -- regresar al ultimo commit algun archivo
git log --ver historial del repositorio
git diff --muestra la diferencia
git branch --lista de todas las ramas
git branch nombre de rama--crear la rama
git checkout "nombre de tu rama"--cambias de rama
git merge nombre de la rama--hacer merge a las otras ramas
